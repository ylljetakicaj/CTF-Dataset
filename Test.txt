private void LoadUsers()
        {
            using (SqlConnection conn = new SqlConnection(connStr))
            using (SqlCommand cmd = new SqlCommand("SELECT DISTINCT useri FROM LOG ORDER BY useri", conn))
            {
                conn.Open();
                SqlDataReader dr = cmd.ExecuteReader();
                ddlUser.Items.Clear();
                ddlUser.Items.Add(new System.Web.UI.WebControls.ListItem("Të gjithë", "all"));

                while (dr.Read())
                {
                    ddlUser.Items.Add(new System.Web.UI.WebControls.ListItem(dr["useri"].ToString(), dr["useri"].ToString()));
                }
                dr.Close();
            }
        }

        private void LoadLogs(string dateRange = "all", string user = "all")
        {
            string query = "SELECT id, id_b, id_api, api, useri, koha, koment FROM LOG WHERE 1=1";

            if (dateRange != "all")
            {
                query += " AND koha >= DATEADD(DAY, -@days, GETDATE())";
            }

            if (user != "all")
            {
                query += " AND useri = @user";
            }

            query += " ORDER BY koha DESC";

            using (SqlConnection conn = new SqlConnection(connStr))
            using (SqlCommand cmd = new SqlCommand(query, conn))
            {
                if (dateRange != "all")
                    cmd.Parameters.AddWithValue("@days", Convert.ToInt32(dateRange));

                if (user != "all")
                    cmd.Parameters.AddWithValue("@user", user);

                conn.Open();
                SqlDataAdapter da = new SqlDataAdapter(cmd);
                DataTable dt = new DataTable();
                da.Fill(dt);

                GridViewLogs.DataSource = dt;
                GridViewLogs.DataBind();
            }
        }

        protected void btnFilter_Click(object sender, EventArgs e)
        {
            string selectedDateRange = ddlDateRange.SelectedValue;
            string selectedUser = ddlUser.SelectedValue;

            LoadLogs(selectedDateRange, selectedUser);
        }
